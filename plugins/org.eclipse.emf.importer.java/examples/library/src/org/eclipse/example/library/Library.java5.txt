package org.eclipse.example.library;
import java.util.List;

/**
 * @model
 */
public interface Library
{
  /**
   * @model
   */
  String getName();

  /**
   * @model type="Writer" containment="true"
   */
  List<Writer> getWriters();

  /**
   * @model type="Book" containment="true"
   */
  List<Book> getBooks();
}
